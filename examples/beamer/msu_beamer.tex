% Before you ask, changing the aspect ratio in the style file is complicated.
% It is better to give you the control over it.
\documentclass[aspectratio=169]{beamer} % add `handout' to options to flatten for students/printing

% \usetheme[logos]{MSU}
\usetheme[logos]{NISL_MSU} % Using MSU theme for no NISL logo; use logos to add logos

% this tells us where to find all the graphics
\graphicspath{ {./images/} }

% This uses biblatex to compile our bibliography.
% You might have to install this.
% It can also break if you use natbib at all (I'm looking at you JMLR)
\usepackage{biblatex}
\addbibresource{ref.bib} % This file is the bib.tex that we are using

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\title{Your Title}
\subtitle{A Subtitle}
\author{Your name}
\institute{Montana State University\\ Your Department}
\date{\today}

% Adds table of contents at beginning of sections
\AtBeginSection[]
{
  \begin{frame}
    \frametitle{Table of Contents}
    \tableofcontents[currentsection]
  \end{frame}
}

\begin{document}

\begin{frame}
	\titlepage{}
\end{frame}

\begin{frame}{Table of Contents}
	\tableofcontents
\end{frame}
